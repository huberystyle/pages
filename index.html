使用方法：

$("seletor").page({
    showLineCount:整数 // 每页显示条数
    showMaxPages:整数 // 默认为7 最小为7
    bPrevNext:boolean // 是否显示上一页 和 下一页 按钮
    bPagination:boolean // 是否显示页码
    effectTime:整数 // 翻页的速度	
})
